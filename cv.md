function getNthFibonacci(n) {
  const lastTwo = [0, 1];
  let counter = 3;
  
  while (counter <= n) {
    const nextFib = lastTwo[0] + lastTwo[1];
    lastTwo[0] = lastTwo[1];

    counter++;
  }

}
